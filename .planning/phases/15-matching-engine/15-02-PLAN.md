class SemanticMatcher:
    """Calculate semantic similarity between JD text and occupational group definitions."""

    def __init__(self):
        # Use all-MiniLM-L6-v2: fast, good accuracy, no GPU needed (per RESEARCH.md)
        self.model = SentenceTransformer('all-MiniLM-L6-v2')
        self._definition_cache = {}  # Cache embeddings for group definitions

    def compute_similarity(self, jd_text: str, definition: str) -> float:
        """Calculate cosine similarity between JD and definition."""

    def shortlist_by_definition(
        self,
        jd_text: str,
        groups: List[Dict],
        min_similarity: float = 0.3
    ) -> List[Dict]:
        """
        Return groups with semantic similarity >= threshold.

        Args:
            jd_text: Combined Client-Service Results + Key Activities text
            groups: List of group dicts with 'definition' key
            min_similarity: Threshold (default 0.3 per CONTEXT.md)

        Returns:
            List of dicts with: group, semantic_similarity, ranked by similarity desc
        """
```

**shortlist_candidates() function:**
Convenience function that:
1. Takes jd_text and groups list
2. Uses SemanticMatcher to compute similarities
3. Also computes similarity to inclusions if available (secondary boost signal)
4. Returns top 10 candidates sorted by combined_score

Per RESEARCH.md Pattern 2:
- Definition similarity is primary signal (80% weight)
- Inclusion similarity is secondary boost (20% weight)
- Combined score = def_similarity * 0.8 + max_inclusion_similarity * 0.2

Cache definition embeddings since groups don't change often. JD embeddings computed fresh each call.

        Option A: If labels contain group code references, extract them
        Option B: Use semantic similarity between labels and group definitions
        Option C: Return empty mapping if no clear connection exists

        For v4.0, use Option C - return 0 boost if we can't establish mapping.
        Future: Add explicit TBS group -> OaSIS code mapping table.
        """

    def get_boost(self, group_code: str, jd_text: str) -> float:

        Returns 0.0 if:
        - No labels mapping established

def shortlist_with_all_signals(
    jd_text: str,
    groups: List[Dict],
    min_similarity: float = 0.3,

) -> List[Dict]:
    """
    Shortlist candidates using all available signals.

    1. Semantic similarity to definition (primary)
    2. Semantic similarity to inclusions (secondary)
    3. Labels.csv boost (if mapping available)

    Args:
        jd_text: Combined Client-Service Results + Key Activities
        groups: List of group dicts from repository (with inclusions)
        min_similarity: Threshold (default 0.3 per CONTEXT.md)
        max_candidates: Max candidates to return (default 10)

    Returns:
        List of dicts sorted by combined_score desc:
        {
            'group': original group dict,
            'semantic_similarity': float,
            'inclusion_similarity': float,
            'labels_boost': float,
            'combined_score': float
        }
    """
```

Note: Labels boost is additive (0.0-0.1), not multiplicative. This ensures even without labels, good semantic matches still rank high.
