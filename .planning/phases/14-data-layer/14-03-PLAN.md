Create `src/scrapers/validation.py` with validation functions per DAMA-DMBOK 2.0:

1. **validate_completeness(data: dict, required_fields: List[str]) -> List[str]:**
   - Check all required_fields exist and are non-empty
   - Return list of error messages (empty if valid)
   - Example errors: "Missing required field: group_code", "Empty value for: definition"

2. **validate_consistency(groups: List[dict]) -> List[str]:**
   - Check no duplicate group_code + subgroup combinations
   - Check all provenance_ids reference valid provenance records
   - Check inclusions/exclusions have valid group references
   - Return list of errors

3. **validate_accuracy(data: dict) -> List[str]:**
   - Check group_code format (2-4 uppercase letters)
   - Check URLs are valid format (start with http:// or https://)
   - Check timestamps are ISO 8601 format
   - Return list of errors

4. **validate_group(group: dict) -> List[str]:**
   Convenience function combining all validations:
   - validate_completeness with ['group_code', 'definition']
   - validate_accuracy on all fields
   - Returns combined error list

5. **ValidationError exception class:**
   - Custom exception with errors list attribute
   - __str__ returns formatted error messages

6. **validate_or_raise(groups: List[dict]):**
   - Runs all validations
   - Raises ValidationError if any errors
   - Returns None if all valid
   - Per CONTEXT.md: "Never insert partial or corrupt data"
  </action>

   ```python
   def scrape_occupational_groups(self) -> dict:
       """
       Full ETL pipeline for occupational groups.
       Returns dict with stats: groups_loaded, inclusions_loaded, etc.
       """
       from datetime import datetime, timezone
       from src.storage.db_manager import get_db, init_db
       from src.storage.repository import OccupationalGroupRepository
       from src.scrapers.html_archiver import archive_html, content_changed
       from src.scrapers.tbs_parser import (
           parse_occupational_groups_table,
           parse_definition_page
       )
       from src.scrapers.validation import validate_or_raise

       # Capture single timestamp for entire scrape (RESEARCH.md Pitfall 6)
       scrape_timestamp = datetime.now(timezone.utc).isoformat()

       # Initialize database if needed
       init_db()

       # EXTRACT: Fetch HTML from TBS
       print("Fetching occupational groups table...")
       table_html, table_meta = self.client.fetch_occupational_groups_table()

       print("Fetching definitions page...")
       defn_html, defn_meta = self.client.fetch_definitions_page()

       # Archive HTML for provenance
       table_archive = archive_html(table_html, table_meta['url'], scrape_timestamp)
       defn_archive = archive_html(defn_html, defn_meta['url'], scrape_timestamp)

       # TRANSFORM: Parse HTML to structured data
       print("Parsing table...")
       table_groups = parse_occupational_groups_table(table_html)

       print("Parsing definitions...")
       definitions = parse_definition_page(defn_html)

       # Merge table data with definitions
       merged_groups = self._merge_table_and_definitions(table_groups, definitions)

       # VALIDATE: Check all data before loading
       print("Validating data...")
       validate_or_raise(merged_groups)

       # LOAD: Atomic transaction
       stats = {'groups': 0, 'inclusions': 0, 'exclusions': 0, 'concordance': 0}

       with get_db() as conn:
           repo = OccupationalGroupRepository(conn)

           # Check if content changed
           if not content_changed(table_meta['url'], table_html, repo):
               print("Content unchanged, skipping load")
               return {'skipped': True, 'reason': 'content_unchanged'}

           try:
               # Insert provenance records
               table_prov_id = repo.insert_provenance({
                   'url': table_meta['url'],
                   'scraped_at': scrape_timestamp,
                   'http_status': table_meta['status'],
                   'http_headers': json.dumps(table_meta['headers']),
                   'content_hash': table_archive['content_hash'],
                   'archive_path': table_archive['archive_path'],
                   'parser_version': PARSER_VERSION
               })

               defn_prov_id = repo.insert_provenance({
                   'url': defn_meta['url'],
                   'scraped_at': scrape_timestamp,
                   'http_status': defn_meta['status'],
                   'http_headers': json.dumps(defn_meta['headers']),
                   'content_hash': defn_archive['content_hash'],
                   'archive_path': defn_archive['archive_path'],
                   'parser_version': PARSER_VERSION
               })

               # Insert groups with inclusions/exclusions
               for group in merged_groups:
                   group_id = repo.insert_group({
                       'group_code': group['group_code'],
                       'subgroup': group.get('subgroup'),
                       'definition': group['definition'],
                       'qualification_standard_url': group.get('qualification_standard_url'),
                       'rates_of_pay_represented_url': group.get('rates_of_pay_represented_url'),
                       'rates_of_pay_unrepresented_url': group.get('rates_of_pay_unrepresented_url'),
                       'effective_from': scrape_timestamp
                   }, defn_prov_id)
                   stats['groups'] += 1

                   # Insert inclusions
                   for incl in group.get('inclusions', []):
                       repo.insert_inclusion(
                           group_id,
                           incl['statement'],
                           incl['order'],
                           defn_prov_id,
                           f"I{incl['order']}"
                       )
                       stats['inclusions'] += 1

                   # Insert exclusions
                   for excl in group.get('exclusions', []):
                       repo.insert_exclusion(
                           group_id,
                           excl['statement'],
                           excl['order'],
                           defn_prov_id,
                           f"E{excl['order']}"
                       )
                       stats['exclusions'] += 1

                   # Insert concordance (job evaluation standard link)
                   if group.get('job_evaluation_standard_url'):
                       repo.insert_concordance(
                           group['group_code'],
                           group['job_evaluation_standard_url'],
                           table_prov_id,
                           scrape_timestamp
                       )
                       stats['concordance'] += 1

               conn.commit()
               print(f"Loaded: {stats}")
               return stats

           except Exception as e:
               conn.rollback()
               print(f"Transaction failed, rolled back: {e}")
               raise
   ```

   ```python
   """
   CLI command to refresh DIM_OCCUPATIONAL table from TBS sources.

   Usage:
       python -m src.cli.refresh_occupational

   This script:
   1. Scrapes TBS occupational groups table and definitions
   2. Validates all data per DAMA-DMBOK 2.0
   3. Archives raw HTML for provenance
   4. Loads to SQLite database (atomic transaction)
   5. Reports statistics

   Rate limited to 1 request/second as per government site etiquette.
   """

   import sys
   import argparse
   from datetime import datetime

   def main():
       parser = argparse.ArgumentParser(
           description='Refresh DIM_OCCUPATIONAL table from TBS sources'
       )
       parser.add_argument(
           '--dry-run',
           action='store_true',
           help='Validate and parse without loading to database'
       )
       parser.add_argument(
           '--verbose', '-v',
           action='store_true',
           help='Show detailed progress'
       )

       args = parser.parse_args()

       print(f"=== TBS Occupational Data Refresh ===")
       print(f"Started: {datetime.now().isoformat()}")
       print()

       try:
           if args.dry_run:
               print("DRY RUN MODE - No database changes")
               print()

               from src.scrapers.http_client import TBSHttpClient
               from src.scrapers.tbs_parser import (
                   parse_occupational_groups_table,
                   parse_definition_page
               )
               from src.scrapers.validation import validate_or_raise

               client = TBSHttpClient()

               print("Fetching pages...")
               table_html, _ = client.fetch_occupational_groups_table()
               defn_html, _ = client.fetch_definitions_page()

               print("Parsing...")
               table_groups = parse_occupational_groups_table(table_html)
               definitions = parse_definition_page(defn_html)

               print("Validating...")
               validate_or_raise(definitions)

               print()
               print(f"Would load:")
               print(f"  - {len(table_groups)} groups from table")
               print(f"  - {len(definitions)} definitions")
               total_incl = sum(len(d.get('inclusions', [])) for d in definitions)
               total_excl = sum(len(d.get('exclusions', [])) for d in definitions)
               print(f"  - {total_incl} inclusions")
               print(f"  - {total_excl} exclusions")
               print()
               print("Validation PASSED")

           else:
               from src.scrapers.tbs_scraper import scrape_all_occupational_groups

               print("Running full scrape...")
               stats = scrape_all_occupational_groups()

               if stats.get('skipped'):
                   print(f"Skipped: {stats.get('reason')}")
               else:
                   print()
                   print("=== Results ===")
                   print(f"Groups loaded: {stats.get('groups', 0)}")
                   print(f"Inclusions loaded: {stats.get('inclusions', 0)}")
                   print(f"Exclusions loaded: {stats.get('exclusions', 0)}")
                   print(f"Concordance entries: {stats.get('concordance', 0)}")

           print()
           print(f"Completed: {datetime.now().isoformat()}")
           return 0

       except Exception as e:
           print(f"ERROR: {e}", file=sys.stderr)
           if args.verbose:
               import traceback
               traceback.print_exc()
           return 1


   if __name__ == '__main__':
       sys.exit(main())
   ```
