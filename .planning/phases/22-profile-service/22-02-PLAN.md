**2. Add renderSourceBadge() function to accordion.js:**

Add near the top of accordion.js (after the constants, before renderOverviewContent):

```javascript
/**
 * Render a source badge indicating whether tab data comes from JobForge parquet or OASIS.
 * @param {string} dataSource - "jobforge" or "oasis"
 * @param {string} [tooltip] - Optional tooltip text for hover
 * @returns {string} HTML string for the badge container
 */
const renderSourceBadge = (dataSource, tooltip) => {
    const label = dataSource === 'jobforge' ? 'Source: JobForge' : 'Source: OASIS';
    const cssClass = dataSource === 'jobforge'
        ? 'source-badge source-badge--jobforge'
        : 'source-badge source-badge--oasis';
    const titleAttr = tooltip ? ` title="${escapeHtml(tooltip)}"` : '';
    return `<div class="source-badge-container">
        <span class="${cssClass}"${titleAttr}>${label}</span>
    </div>`;
};
```

    """NOC source metadata for compliance appendix."""
    noc_code: str

**2. Extend build_compliance_sections() in src/services/export_service.py:**

Modify the Section 6.2.3 (Data Sources) compliance section to include per-section provenance when section_sources is present.

In `build_compliance_sections()`, after building the existing source_content dict:

```python
# Section 6.2.3: Data Sources
source_content = {
    "data_steward": "Employment and Social Development Canada (ESDC)",
    "authoritative_source": "National Occupational Classification (NOC)",
    "access_method": "Direct retrieval from OASIS and JobForge 2.0 gold parquet",  # UPDATE this string
    "source_url": request.source_metadata.profile_url,
    "retrieval_timestamp": request.source_metadata.scraped_at.isoformat(),
    "noc_version": request.source_metadata.version,
}

# Add per-section source provenance if available (PROF-03)
if request.source_metadata.section_sources:
    source_content["section_sources"] = {}
    for section_key, source_type in request.source_metadata.section_sources.items():
        if source_type == "jobforge":
            detail = f"JobForge 2.0 gold parquet (oasis_{section_key}.parquet)"
        else:
            detail = f"OASIS live scraping ({request.source_metadata.profile_url})"
        source_content["section_sources"][section_key] = {
            "source": source_type,
            "detail": detail,
        }
```

Also update the `access_method` value:
- When section_sources is NOT present (old exports): keep "Direct retrieval from OASIS (Occupational and Skills Information System)"
- When section_sources IS present (new exports): use "Direct retrieval from OASIS and JobForge 2.0 gold parquet"

This conditional avoids changing existing export output for pre-Phase 22 exports.
